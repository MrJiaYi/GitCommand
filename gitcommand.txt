GitCommand

	git init
	#初始化git仓库#
	
	git add <filename>
	#添加文件到仓库，可以添加多个文件#
	
	git commit -m “info”
	#提交文件到仓库#
	
	git status
	#查看工作区状态#
	
	git log
	#查看工作日志#
	git log --pretty=oneline
	#查看工作日志-简化单行版#
	
	git reflog
	#查看命令历史#
	
	git checkout -- filename
	#丢弃工作区的修改（没有add）#
	
	git reset HEAD filename 
	#被添加到暂存区（已经add过）丢弃修改，也就是直接返回上一条，可以继续实用checkout继续撤销#
	
	git reset --hard commit_id
	#已经commit，回退到之前版本，需要版本id，查看id用gitlog或者git reflog#

	git rm filename
	#删除文件#
